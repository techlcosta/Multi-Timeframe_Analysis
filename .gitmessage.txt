# <type>(<scope>): <subject>
#
# Examples:
# feat(api): add structured MT5 error responses
# fix(ui): prevent pywebview callback crash on API failure
# docs(readme): rewrite project overview in three languages
# ci(release): build Windows binary on version tags
#
# Allowed types:
# feat, fix, docs, style, refactor, perf, test, build, ci, chore, revert
#
# Rules:
# - use lowercase type
# - optional scope in kebab-case
# - concise subject after ": "
# - no trailing period
